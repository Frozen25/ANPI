269 void Band_matrix::lu_decompose()
270 {
271  int i_max,j_max;
272  int j_min;
273  float x;
274 
275  // preconditioning
276  // normalize column i so that a_ii=1
277  for(int i=0; i<this->dim(); i++) {
278  assert(this->operator()(i,i)!=0.0);
279  this->saved_diag(i)=1.0/this->operator()(i,i);
280  j_min=std::max(0,i-this->num_lower());
281  j_max=std::min(this->dim()-1,i+this->num_upper());
282  for(int j=j_min; j<=j_max; j++) {
283  this->operator()(i,j) *= this->saved_diag(i);
284  }
285  this->operator()(i,i)=1.0; // prevents rounding errors
286  }
287 
288  // Gauss LR-Decomposition
289  for(int k=0; k<this->dim(); k++) {
290  i_max=std::min(this->dim()-1,k+this->num_lower()); // num_lower not a mistake!
291  for(int i=k+1; i<=i_max; i++) {
292  assert(this->operator()(k,k)!=0.0);
293  x=-this->operator()(i,k)/this->operator()(k,k);
294  this->operator()(i,k)=-x; // assembly part of L
295  j_max=std::min(this->dim()-1,k+this->num_upper());
296  for(int j=k+1; j<=j_max; j++) {
297  // assembly part of R
298  this->operator()(i,j)=this->operator()(i,j)+x*this->operator()(k,j);
299  }
300  }
301  }
302 }

310 std::vector<float> Band_matrix::l_solve(const std::vector<float>& b) const
311 {
312  assert( this->dim()==(int)b.size() );
313  std::vector<float> x(this->dim());
314  int j_start;
315  float sum;
316  for(int i=0; i<this->dim(); i++) {
317  sum=0;
318  j_start=std::max(0,i-this->num_lower());
319  for(int j=j_start; j<i; j++) sum += this->operator()(i,j)*x[j];
320  x[i]=(b[i]*this->saved_diag(i)) - sum;
321  }
322  return x;
323 }

331 std::vector<float> Band_matrix::r_solve(const std::vector<float>& b) const
332 {
333  assert( this->dim()==(int)b.size() );
334  std::vector<float> x(this->dim());
335  int j_stop;
336  float sum;
337  for(int i=this->dim()-1; i>=0; i--) {
338  sum=0;
339  j_stop=std::min(this->dim()-1,i+this->num_upper());
340  for(int j=i+1; j<=j_stop; j++) sum += this->operator()(i,j)*x[j];
341  x[i]=( b[i] - sum ) / this->operator()(i,i);
342  }
343  return x;
344 }

398 void spline::set_points(const std::vector<float>& x,
399  const std::vector<float>& y, bool cubic_spline)
400 {
401  assert(x.size()==y.size());
402  assert(x.size()>2);
403  m_x=x;
404  m_y=y;
405  int n=x.size();
406  // TODO: maybe sort x and y, rather than returning an error
407  for(int i=0; i<n-1; i++) {
408  assert(m_x[i]<m_x[i+1]);
409  }
410 
411  if(cubic_spline==true) { // cubic spline interpolation
412  // setting up the matrix and right hand side of the equation system
413  // for the parameters b[]
414  Band_matrix A(n,1,1);
415  std::vector<float> rhs(n);
416  for(int i=1; i<n-1; i++) {
417  A(i,i-1)=1.0/3.0*(x[i]-x[i-1]);
418  A(i,i)=2.0/3.0*(x[i+1]-x[i-1]);
419  A(i,i+1)=1.0/3.0*(x[i+1]-x[i]);
420  rhs[i]=(y[i+1]-y[i])/(x[i+1]-x[i]) - (y[i]-y[i-1])/(x[i]-x[i-1]);
421  }
422  // boundary conditions
423  if(m_left == spline::second_deriv) {
424  // 2*b[0] = f''
425  A(0,0)=2.0;
426  A(0,1)=0.0;
427  rhs[0]=m_left_value;
428  } else if(m_left == spline::first_deriv) {
429  // c[0] = f', needs to be re-expressed in terms of b:
430  // (2b[0]+b[1])(x[1]-x[0]) = 3 ((y[1]-y[0])/(x[1]-x[0]) - f')
431  A(0,0)=2.0*(x[1]-x[0]);
432  A(0,1)=1.0*(x[1]-x[0]);
433  rhs[0]=3.0*((y[1]-y[0])/(x[1]-x[0])-m_left_value);
434  } else {
435  assert(false);
436  }
437  if(m_right == spline::second_deriv) {
438  // 2*b[n-1] = f''
439  A(n-1,n-1)=2.0;
440  A(n-1,n-2)=0.0;
441  rhs[n-1]=m_right_value;
442  } else if(m_right == spline::first_deriv) {
443  // c[n-1] = f', needs to be re-expressed in terms of b:
444  // (b[n-2]+2b[n-1])(x[n-1]-x[n-2])
445  // = 3 (f' - (y[n-1]-y[n-2])/(x[n-1]-x[n-2]))
446  A(n-1,n-1)=2.0*(x[n-1]-x[n-2]);
447  A(n-1,n-2)=1.0*(x[n-1]-x[n-2]);
448  rhs[n-1]=3.0*(m_right_value-(y[n-1]-y[n-2])/(x[n-1]-x[n-2]));
449  } else {
450  assert(false);
451  }
452 
453  // solve the equation system to obtain the parameters b[]
454  m_b=A.lu_solve(rhs);
455 
456  // calculate parameters a[] and c[] based on b[]
457  m_a.resize(n);
458  m_c.resize(n);
459  for(int i=0; i<n-1; i++) {
460  m_a[i]=1.0/3.0*(m_b[i+1]-m_b[i])/(x[i+1]-x[i]);
461  m_c[i]=(y[i+1]-y[i])/(x[i+1]-x[i])
462  - 1.0/3.0*(2.0*m_b[i]+m_b[i+1])*(x[i+1]-x[i]);
463  }
464  } else { // linear interpolation
465  m_a.resize(n);
466  m_b.resize(n);
467  m_c.resize(n);
468  for(int i=0; i<n-1; i++) {
469  m_a[i]=0.0;
470  m_b[i]=0.0;
471  m_c[i]=(m_y[i+1]-m_y[i])/(m_x[i+1]-m_x[i]);
472  }
473  }
474 
475  // for left extrapolation coefficients
476  m_b0 = (m_force_linear_extrapolation==false) ? m_b[0] : 0.0;
477  m_c0 = m_c[0];
478 
479  // for the right extrapolation coefficients
480  // f_{n-1}(x) = b*(x-x_{n-1})^2 + c*(x-x_{n-1}) + y_{n-1}
481  float h=x[n-1]-x[n-2];
482  // m_b[n-1] is determined by the boundary condition
483  m_a[n-1]=0.0;
484  m_c[n-1]=3.0*m_a[n-2]*h*h+2.0*m_b[n-2]*h+m_c[n-2]; // = f'_{n-2}(x_{n-1})
485  if(m_force_linear_extrapolation==true)
486  m_b[n-1]=0.0;
487 }

494 float spline::operator() (float x) const
495 {
496  size_t n=m_x.size();
497  // find the closest point m_x[idx] < x, idx=0 even if x<m_x[0]
498  std::vector<float>::const_iterator it;
499  it=std::lower_bound(m_x.begin(),m_x.end(),x);
500  int idx=std::max( int(it-m_x.begin())-1, 0);
501 
502  float h=x-m_x[idx];
503  float interpol;
504  if(x<m_x[0]) {
506  interpol=(m_b0*h + m_c0)*h + m_y[0];
507  } else if(x>m_x[n-1]) {
509  interpol=(m_b[n-1]*h + m_c[n-1])*h + m_y[n-1];
510  } else {
512  interpol=((m_a[idx]*h + m_b[idx])*h + m_c[idx])*h + m_y[idx];
513  }
514  return interpol;
515 }
